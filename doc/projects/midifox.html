#include <MIDI.h>

MIDI_CREATE_DEFAULT_INSTANCE();

// Pin definitions
const int ledPin = 13;
const int startDigitalPin = 5;
const int numButtons = 8; // Digital buttons
const int numAnalogInputs = 4; // Including page buttons
const int numMuxButtons = 8; // Buttons connected to multiplexer
const int muxPin = 10; // Multiplexer output pin

// Multiplexer control pins
const int selectPins[] = {2, 3, 4};

// Page definitions
const byte pages[] = {48, 66, 84, 102}; // CC numbers for pages
int currentPageIndex = 0;

// State arrays
int currentButtonStates[numButtons] = {0};
int previousButtonStates[numButtons] = {0};
int currentAnalogStates[numAnalogInputs] = {0};
int previousAnalogStates[numAnalogInputs] = {0};
int currentMuxStates[numMuxButtons] = {0};
int previousMuxStates[numMuxButtons] = {0};

void setup() {
    MIDI.begin();
    Serial.begin(115200);
    
    pinMode(ledPin, INPUT);
    for (int i = 0; i < numButtons; i++) {
        pinMode(startDigitalPin + i, INPUT_PULLUP);
        previousButtonStates[i] = HIGH; // Not pressed state
    }
    
    for (int i = 0; i < numAnalogInputs; i++) {
        previousAnalogStates[i] = LOW;
    }

    // Set multiplexer control pins as outputs
    for (int pin : selectPins) {
        pinMode(pin, OUTPUT);
    }
}

void loop() {
    updateDigitalButtons();
    updateAnalogButtons();
    updateMultiplexerButtons();
    
    handleLedState();
}

void updateDigitalButtons() {
    for (int i = 0; i < numButtons; i++) {
        currentButtonStates[i] = digitalRead(startDigitalPin + i);
        if (currentButtonStates[i] != previousButtonStates[i]) {
            byte noteNumber = pages[currentPageIndex] + i;
            if (currentButtonStates[i] == LOW) {
                MIDI.sendNoteOn(noteNumber, 127, 1); // Note on
            } else {
                MIDI.sendNoteOff(noteNumber, 0, 1); // Note off
            }
            previousButtonStates[i] = currentButtonStates[i];
        }
    }
}

void updateAnalogButtons() {
    for (int i = 0; i < numAnalogInputs; i++) {
        currentAnalogStates[i] = digitalRead(14 + i); // Assuming analog inputs start from pin 14
        if (currentAnalogStates[i] != previousAnalogStates[i]) {
            if (currentAnalogStates[i] == HIGH) {
                currentPageIndex = i % sizeof(pages); // Cycle through pages
                Serial.print("Page changed to: ");
                Serial.println(currentPageIndex + 1);
            }
            previousAnalogStates[i] = currentAnalogStates[i];
        }
    }
}

void updateMultiplexerButtons() {
    for (int i = 0; i < numMuxButtons; i++) {
        currentMuxStates[i] = getMuxValue(i);
        if (currentMuxStates[i] != previousMuxStates[i]) {
            byte noteNumber = pages[currentPageIndex] + 9 + i;
            if (currentMuxStates[i] == LOW) {
                MIDI.sendNoteOn(noteNumber, 127, 1); // Note on
            } else {
                MIDI.sendNoteOff(noteNumber, 0, 1); // Note off
            }
            previousMuxStates[i] = currentMuxStates[i];
        }
    }
}

int getMuxValue(int index) {
    for (int bit = 0; bit < sizeof(selectPins); bit++) {
        digitalWrite(selectPins[bit], bitRead(index, bit));
    }
    return digitalRead(muxPin);
}

void handleLedState() {
    int ledNoteNumber = pages[currentPageIndex];
    int ledState = digitalRead(ledPin);
    
    static int lastLedState = HIGH;
    
    if (ledState != lastLedState) {
        if (ledState == LOW) {
            MIDI.sendNoteOn(ledNoteNumber, 127, 1); // Note on
        } else {
            MIDI.sendNoteOff(ledNoteNumber, 0, 1); // Note off
        }
        lastLedState = ledState;
    }
}
